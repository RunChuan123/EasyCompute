PyTorch/TensorFlow这类框架底层的“设备抽象层”设计思路。
核心解决思路是：**用「接口分层」+「类型擦除」+「特化实现」**，既保证抽象接口的统一性，又不丢失不同设备（CUDA/CPU）的特有功能。

我直接给你一套**可落地的C++抽象设计方案**，完美适配你的场景（Eager/Trace模式、CUDA流管理、CPU无流）。

---

# 一、核心设计思路（先理解再写代码）
### 1. 三层抽象结构（从顶到下）
| 层级 | 作用 | 包含内容 |
|------|------|----------|
| **DeviceContext（纯虚基类）** | 所有设备的通用接口 | 显存/内存分配、释放、同步（无设备特有逻辑） |
| **GPUContext（NVContext，继承DeviceContext）** | CUDA特有实现 | 流管理、事件、显存池（重写通用接口+扩展特有接口） |
| **CPUContext（继承DeviceContext）** | CPU特有实现 | 内存分配（重写通用接口，无流/事件） |

### 2. 关键原则
- **通用逻辑上提**：分配/释放/同步等所有设备都有的操作，放在基类
- **特有逻辑下沉**：CUDA的流/事件、CPU的内存对齐等，在子类扩展
- **管理器只认基类**：DeviceManager统一管理DeviceContext*，不用关心具体是CPU/GPU

# 二、完整代码实现（你直接拷贝用）
## 第一步：定义抽象基类 DeviceContext
```cpp
#include <memory>
#include <mutex>
#include <unordered_map>
#include <string>
#include <cstdint>

// 前向声明：CUDA相关类型（避免CPU编译依赖CUDA）
#ifdef CUDA_ENABLED
struct cudaStream_t;
struct cudaEvent_t;
#endif

// 设备类型枚举（框架全局通用）
enum class DeviceType {
    CPU,
    CUDA
};

// ===================== 核心抽象接口 =====================
class DeviceContext {
public:
    // 纯虚析构（必须有）
    virtual ~DeviceContext() = default;

    // ========== 所有设备通用的核心接口 ==========
    // 1. 获取设备类型/ID
    virtual DeviceType type() const = 0;
    virtual int device_id() const = 0;

    // 2. 内存分配/释放（通用接口，子类各自实现）
    virtual void* allocate(size_t size) = 0;
    virtual void deallocate(void* ptr) = 0;

    // 3. 同步（通用语义：等待所有操作完成）
    virtual void sync() = 0;

    // ========== 设备特有接口（类型安全的向下转换） ==========
    // CUDA特有接口：子类实现，基类返回nullptr
    virtual cudaStream_t get_compute_stream(int stream_idx = 0) { return nullptr; }
    virtual cudaStream_t get_h2d_stream() { return nullptr; }
    virtual cudaStream_t get_mem_stream() { return nullptr; }
    virtual void record_event(cudaStream_t stream, cudaEvent_t& event) {}
    virtual void wait_event(cudaStream_t stream, cudaEvent_t event) {}

    // CPU特有接口：子类实现，基类空实现
    virtual void set_cpu_affinity(int core_id) {}
};
```

## 第二步：实现 CUDA 上下文（NVContext）
```cpp
#ifdef CUDA_ENABLED
#include <cuda_runtime.h>

// ===================== CUDA上下文实现 =====================
class NVContext : public DeviceContext {
private:
    // CUDA特有成员（流池、显存池、事件等）
    int device_id_ = -1;
    cudaDeviceProp prop_;
    void* mem_pool_ = nullptr;
    
    // 流池：计算流（多个）、H2D流、内存流
    std::vector<cudaStream_t> compute_streams_;
    cudaStream_t h2d_stream_ = nullptr;
    cudaStream_t mem_stream_ = nullptr;
    
    // 线程安全锁
    std::mutex mutex_;

    // 私有构造（单例）
    NVContext(int device_id) : device_id_(device_id) {
        // 初始化CUDA设备、流、显存池（复用之前的逻辑）
        cudaSetDevice(device_id_);
        cudaGetDeviceProperties(&prop_, device_id_);

        // 1. 创建计算流池（4个足够吃满GPU）
        for (int i = 0; i < 4; ++i) {
            cudaStream_t stream;
            cudaStreamCreateWithFlags(&stream, cudaStreamNonBlocking);
            compute_streams_.push_back(stream);
        }

        // 2. 创建H2D流、内存流
        cudaStreamCreateWithFlags(&h2d_stream_, cudaStreamNonBlocking);
        cudaStreamCreateWithFlags(&mem_stream_, cudaStreamNonBlocking);

        // 3. 初始化显存池（CUDA 11.2+）
        if (cudaRuntimeGetVersion() >= 11020) {
            cudaMemPoolProps pool_props{};
            pool_props.allocType = cudaMemAllocationTypePinned;
            pool_props.handleTypes = cudaMemHandleTypeNone;
            cudaDeviceGetDefaultMemPool(&mem_pool_, device_id_);
            cudaMemPoolSetAttribute(mem_pool_, cudaMemPoolAttrReleaseThreshold, 0);
        }
    }

public:
    // 单例获取（CUDA上下文唯一）
    static std::shared_ptr<NVContext> get_instance(int device_id) {
        static std::unordered_map<int, std::shared_ptr<NVContext>> ctx_map;
        static std::mutex global_mutex;
        std::lock_guard<std::mutex> lock(global_mutex);
        if (!ctx_map.count(device_id)) {
            ctx_map[device_id] = std::shared_ptr<NVContext>(new NVContext(device_id));
        }
        return ctx_map[device_id];
    }

    // ========== 重写基类通用接口 ==========
    DeviceType type() const override { return DeviceType::CUDA; }
    int device_id() const override { return device_id_; }

    void* allocate(size_t size) override {
        std::lock_guard<std::mutex> lock(mutex_);
        cudaSetDevice(device_id_);
        void* ptr = nullptr;
        if (mem_pool_) {
            cudaMallocFromPoolAsync(&ptr, size, mem_pool_, mem_stream_);
        } else {
            cudaMallocAsync(&ptr, size, mem_stream_);
        }
        return ptr;
    }

    void deallocate(void* ptr) override {
        if (!ptr) return;
        std::lock_guard<std::mutex> lock(mutex_);
        cudaSetDevice(device_id_);
        cudaFreeAsync(ptr, mem_stream_);
    }

    void sync() override {
        std::lock_guard<std::mutex> lock(mutex_);
        cudaSetDevice(device_id_);
        // 同步所有流
        for (auto stream : compute_streams_) {
            cudaStreamSynchronize(stream);
        }
        cudaStreamSynchronize(h2d_stream_);
        cudaStreamSynchronize(mem_stream_);
    }

    // ========== 重写CUDA特有接口（核心） ==========
    cudaStream_t get_compute_stream(int stream_idx = 0) override {
        std::lock_guard<std::mutex> lock(mutex_);
        if (stream_idx < 0 || stream_idx >= compute_streams_.size()) {
            return compute_streams_[0]; // 默认返回第一个计算流
        }
        return compute_streams_[stream_idx];
    }

    cudaStream_t get_h2d_stream() override {
        return h2d_stream_;
    }

    cudaStream_t get_mem_stream() override {
        return mem_stream_;
    }

    void record_event(cudaStream_t stream, cudaEvent_t& event) override {
        std::lock_guard<std::mutex> lock(mutex_);
        cudaSetDevice(device_id_);
        cudaEventRecord(event, stream);
    }

    void wait_event(cudaStream_t stream, cudaEvent_t event) override {
        std::lock_guard<std::mutex> lock(mutex_);
        cudaSetDevice(device_id_);
        cudaStreamWaitEvent(stream, event, 0);
    }

    // 析构：释放流
    ~NVContext() override {
        std::lock_guard<std::mutex> lock(mutex_);
        if (device_id_ == -1) return;
        // 释放计算流池
        for (auto stream : compute_streams_) {
            cudaStreamDestroy(stream);
        }
        cudaStreamDestroy(h2d_stream_);
        cudaStreamDestroy(mem_stream_);
        // 释放显存池（CUDA内部自动管理）
        mem_pool_ = nullptr;
    }

    // 禁用拷贝
    NVContext(const NVContext&) = delete;
    NVContext& operator=(const NVContext&) = delete;
};
#endif // CUDA_ENABLED
```

## 第三步：实现 CPU 上下文（CPUContext）
```cpp
// ===================== CPU上下文实现 =====================
class CPUContext : public DeviceContext {
private:
    int device_id_ = 0; // CPU默认设备ID为0（多核用core_id区分）
    std::mutex mutex_;

    // 私有构造（单例）
    CPUContext(int device_id) : device_id_(device_id) {}

public:
    // 单例获取
    static std::shared_ptr<CPUContext> get_instance(int device_id = 0) {
        static std::unordered_map<int, std::shared_ptr<CPUContext>> ctx_map;
        static std::mutex global_mutex;
        std::lock_guard<std::mutex> lock(global_mutex);
        if (!ctx_map.count(device_id)) {
            ctx_map[device_id] = std::shared_ptr<CPUContext>(new CPUContext(device_id));
        }
        return ctx_map[device_id];
    }

    // ========== 重写基类通用接口 ==========
    DeviceType type() const override { return DeviceType::CPU; }
    int device_id() const override { return device_id_; }

    void* allocate(size_t size) override {
        std::lock_guard<std::mutex> lock(mutex_);
        // CPU内存分配（可加内存对齐优化）
        return malloc(size);
    }

    void deallocate(void* ptr) override {
        if (!ptr) return;
        std::lock_guard<std::mutex> lock(mutex_);
        free(ptr);
    }

    void sync() override {
        // CPU无异步操作，同步为空
    }

    // ========== 重写CPU特有接口 ==========
    void set_cpu_affinity(int core_id) override {
        // 可选：设置CPU核心绑定（多线程优化）
    }

    // 禁用拷贝
    CPUContext(const CPUContext&) = delete;
    CPUContext& operator=(const CPUContext&) = delete;
};
```

## 第四步：实现全局 DeviceManager（统一管理）
```cpp
// ===================== 全局设备管理器 =====================
class DeviceManager {
private:
    // 线程局部存储：当前线程的默认设备上下文
    static thread_local std::shared_ptr<DeviceContext> current_ctx_;

public:
    // 1. 设置当前设备（自动创建对应上下文）
    static void set_current_device(DeviceType type, int device_id = 0) {
        if (type == DeviceType::CUDA) {
#ifdef CUDA_ENABLED
            current_ctx_ = NVContext::get_instance(device_id);
#else
            throw std::runtime_error("CUDA is not enabled");
#endif
        } else if (type == DeviceType::CPU) {
            current_ctx_ = CPUContext::get_instance(device_id);
        }
    }

    // 2. 获取当前设备上下文（基类指针，统一接口）
    static std::shared_ptr<DeviceContext> get_current_context() {
        if (!current_ctx_) {
            // 默认使用CPU上下文
            current_ctx_ = CPUContext::get_instance(0);
        }
        return current_ctx_;
    }

    // 3. 类型安全的向下转换（获取特有上下文）
    template <typename T>
    static std::shared_ptr<T> get_specific_context() {
        auto base_ctx = get_current_context();
        return std::dynamic_pointer_cast<T>(base_ctx);
    }
};

// 初始化线程局部存储
thread_local std::shared_ptr<DeviceContext> DeviceManager::current_ctx_ = nullptr;
```

# 三、核心用法示例（框架里怎么用）
### 1. 通用场景（不区分设备）
```cpp
// 设置当前设备为CPU
DeviceManager::set_current_device(DeviceType::CPU);
auto ctx = DeviceManager::get_current_context();
// 分配内存（自动调用CPUContext的allocate）
void* cpu_data = ctx->allocate(1024);
ctx->deallocate(cpu_data);

// 切换到CUDA设备0
DeviceManager::set_current_device(DeviceType::CUDA, 0);
ctx = DeviceManager::get_current_context();
// 分配显存（自动调用NVContext的allocate）
void* cuda_data = ctx->allocate(1024);
ctx->deallocate(cuda_data);
```

### 2. CUDA特有场景（Trace模式流管理）
```cpp
// 获取CUDA特有上下文（类型安全转换）
auto nv_ctx = DeviceManager::get_specific_context<NVContext>();
if (nv_ctx) {
    // 获取计算流、H2D流
    auto comp_stream = nv_ctx->get_compute_stream(0);
    auto h2d_stream = nv_ctx->get_h2d_stream();
    auto mem_stream = nv_ctx->get_mem_stream();

    // 记录事件、等待事件（Trace模式重叠核心）
    cudaEvent_t evt;
    cudaEventCreate(&evt);
    nv_ctx->record_event(h2d_stream, evt);
    nv_ctx->wait_event(comp_stream, evt);

    // 执行kernel（用计算流）
    // my_kernel<<<grid, block, 0, comp_stream>>>(...);
}
```

### 3. Trace模式执行器（自动适配设备）
```cpp
void execute_op(DeviceContext* ctx, void* data, size_t size) {
    // 通用逻辑：分配内存
    void* ptr = ctx->allocate(size);

    // 设备特有逻辑：CUDA需要流/事件，CPU不需要
    if (ctx->type() == DeviceType::CUDA) {
#ifdef CUDA_ENABLED
        auto nv_ctx = static_cast<NVContext*>(ctx);
        auto comp_stream = nv_ctx->get_compute_stream(0);
        // CUDA异步拷贝
        cudaMemcpyAsync(ptr, data, size, cudaMemcpyHostToDevice, nv_ctx->get_h2d_stream());
        // 执行kernel
        // kernel<<<..., comp_stream>>>(ptr);
#endif
    } else {
        // CPU同步拷贝
        memcpy(ptr, data, size);
        // CPU计算
        // cpu_kernel(ptr);
    }
}
```

# 四、关键设计亮点（解决你的核心问题）
### 1. 接口统一 + 特有功能不丢失
- 基类 `DeviceContext` 定义所有设备的通用接口，管理器只认基类；
- 子类通过**重写特有接口**暴露CUDA流/事件（CPU子类空实现），既不破坏抽象，又能用到CUDA的特有功能。

### 2. 类型安全的向下转换
- `DeviceManager::get_specific_context<T>()` 提供类型安全的转换，避免裸指针强制转换的风险；
- 框架在需要CUDA特有功能时，先转换再调用，CPU场景直接忽略。

### 3. 线程安全 + 单例
- 每个设备上下文是单例，避免重复创建；
- 所有设备操作加锁，支持多线程训练/推理。

### 4. 兼容 Eager/Trace 模式
- Eager模式：直接调用 `sync()`，CUDA同步所有流，CPU空操作；
- Trace模式：通过 `get_compute_stream()`/`record_event()` 实现流重叠，CPU无感知。

# 总结
1. **抽象分层**：`DeviceContext`（基类）→ `NVContext`/`CPUContext`（子类），通用逻辑上提，特有逻辑下沉；
2. **接口设计**：基类只留通用接口，子类重写特有接口（CUDA暴露流/事件，CPU空实现）；
3. **管理器封装**：`DeviceManager` 统一管理上下文，提供类型安全的向下转换，框架层无需关心具体设备类型。

这套设计完全适配你的ML框架需求：既保证CPU/CUDA设备的抽象统一，又能充分利用CUDA流/事件做Trace模式的重叠优化，同时兼容Eager模式的简单同步逻辑。

你可以直接基于这套代码扩展（比如添加ROCMContext、TPUContext等），抽象层完全复用。
